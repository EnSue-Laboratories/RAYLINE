### Task 2: GitHub manager module (gh CLI wrapper)

**Files:**
- Create: `electron/github-manager.cjs`

**Step 1: Create the gh CLI wrapper**

Create `electron/github-manager.cjs`. This module wraps all `gh api` calls. Key functions:

```js
const { execFile } = require("child_process");

function gh(args) {
  return new Promise((resolve, reject) => {
    execFile("gh", args, {
      env: { ...process.env, GH_NO_UPDATE_NOTIFIER: "1" },
      timeout: 15000,
      maxBuffer: 5 * 1024 * 1024,
    }, (err, stdout, stderr) => {
      if (err) reject(new Error(stderr.trim() || err.message));
      else resolve(stdout.trim());
    });
  });
}

async function ghJson(args) {
  const raw = await gh(args);
  return raw ? JSON.parse(raw) : null;
}

async function checkAuth() {
  try {
    await gh(["auth", "status"]);
    return { ok: true };
  } catch (e) {
    return { ok: false, error: e.message };
  }
}

async function listUserRepos() {
  return ghJson([
    "api", "/user/repos",
    "--paginate",
    "--jq", ".[].full_name",
    "-q", ".[]",
  ]);
  // Actually: use gh repo list which is simpler
}
```

Provide these exports:

- `checkAuth()` — returns `{ ok, error? }`
- `listUserRepos(limit)` — returns `["owner/repo", ...]` via `gh repo list --json nameWithOwner --limit N`
- `listIssues(repo, state)` — returns issues array via `gh api /repos/{owner}/{repo}/issues?state={state}&per_page=100`
- `listPRs(repo, state)` — returns PRs array via `gh api /repos/{owner}/{repo}/pulls?state={state}&per_page=100`
- `getIssue(repo, number)` — single issue detail
- `getPR(repo, number)` — single PR detail
- `listComments(repo, number)` — issue/PR comments
- `addComment(repo, number, body)` — POST comment
- `listCollaborators(repo)` — repo collaborators
- `assignIssue(repo, number, assignees)` — PATCH assignees
- `unassignIssue(repo, number, assignees)` — remove assignees

Use `gh api` for all calls. The `gh` binary handles auth transparently.

const { contextBridge, ipcRenderer } = require("electron");

contextBridge.exposeInMainWorld("ghApi", {
  checkAuth: () => ipcRenderer.invoke("gh-check-auth"),
  listUserRepos: (limit) => ipcRenderer.invoke("gh-list-user-repos", limit),
  listIssues: (repo, state) => ipcRenderer.invoke("gh-list-issues", repo, state),
  listPRs: (repo, state) => ipcRenderer.invoke("gh-list-prs", repo, state),
  getIssue: (repo, number) => ipcRenderer.invoke("gh-get-issue", repo, number),
  getPR: (repo, number) => ipcRenderer.invoke("gh-get-pr", repo, number),
  listComments: (repo, number) => ipcRenderer.invoke("gh-list-comments", repo, number),
  addComment: (repo, number, body) => ipcRenderer.invoke("gh-add-comment", repo, number, body),
  listCollaborators: (repo) => ipcRenderer.invoke("gh-list-collaborators", repo),
  assignIssue: (repo, number, assignees) => ipcRenderer.invoke("gh-assign-issue", repo, number, assignees),
  unassignIssue: (repo, number, assignees) => ipcRenderer.invoke("gh-unassign-issue", repo, number, assignees),
  // State persistence (reuse existing state file)

function createProjectManagerWindow() {

    width: 1000,
    height: 700,

    minHeight: 500,
    titleBarStyle: "hiddenInset",
    trafficLightPosition: { x: 16, y: 18 },
    backgroundColor: "#000000",

    webPreferences: {
      preload: path.join(__dirname, "preload-pm.cjs"),
      contextIsolation: true,
      nodeIntegration: false,
    },
  });

  if (isDev) {
    const port = process.env.VITE_PORT || "5173";
    pmWindow.loadURL(`http://localhost:${port}/src/project-manager.html`);
  } else {
    pmWindow.loadFile(path.join(__dirname, "../dist/src/project-manager.html"));

Add PM state handlers that read/write `pmRepos` to `claudi-state.json`:

```js
ipcMain.handle("gh-load-pm-state", async () => {
  try {
    if (fs.existsSync(stateFilePath)) {
      const data = JSON.parse(fs.readFileSync(stateFilePath, "utf-8"));
      return { repos: data.pmRepos || [] };
    }
  } catch {}
  return { repos: [] };
});

ipcMain.handle("gh-save-pm-state", async (_e, pmState) => {
  try {
    let data = {};
    if (fs.existsSync(stateFilePath)) {
      data = JSON.parse(fs.readFileSync(stateFilePath, "utf-8"));
    }
    data.pmRepos = pmState.repos || [];
    fs.writeFileSync(stateFilePath, JSON.stringify(data, null, 2));
    return true;
  } catch { return false; }
});
```
